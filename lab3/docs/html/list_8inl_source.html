28template <typename T> void List<T>::ensure_capacity(int desired) {
29 if (this == nullptr) {
30 Logger::critical("Attempted to extend a null list!");
31 }
32 if (desired <= capacity_) {
33 return;
34 }
35
36 int new_capacity = (capacity_ == 0) ? 4 : capacity_;
37 while (new_capacity < desired) {
38 new_capacity *= 2;
39 }
40
41 std::unique_ptr<T[]> new_data = std::make_unique<T[]>(new_capacity);
42
43 // Only attempt to move if we actually have existing data
44 if (data_) {
45 for (int i = 0; i < size_; ++i) {
46 new_data[i] = std::move(data_[i]);
47 }
48 }
49
50 data_ = std::move(new_data);
51 capacity_ = new_capacity;
52}

104template <typename T> std::unique_ptr<Container<T>> List<T>::copy() const {
105 auto newList = std::make_unique<List<T>>();
106 newList->ensure_capacity(this->size_);
107
108 for (int i = 0; i < this->size_; ++i) {
109 if (!data_[i]) {
110 newList->append(nullptr);
111 continue;
112 }
113
114 if constexpr (is_unique_ptr<T>::value) {
115 using ElementType = typename T::element_type;
116
117 // Polymorphic Clone
118 if constexpr (requires(const ElementType &e) { e.copy(); }) {
119 newList->append(data_[i]->copy());
120 }
121 // Try to use copy constructor if it exists
122 else if constexpr (std::is_copy_constructible_v<ElementType>) {
123 newList->append(std::make_unique<ElementType>(*data_[i]));
124 } else {
125 // Error out if not able to copy
126 static_assert(std::is_copy_constructible_v<ElementType>,
127 "Type inside unique_ptr must have a .copy() method or a "
128 "copy constructor.");
129 }
130 } else {
131 // Standard non-pointer types
132 newList->append(data_[i]);
133 }
134 }
135 return newList;
136}

145template <typename T> void List<T>::slice(int min, int max) {
146 // Bounds checking and validation
147 if (min < 0)
148 min = 0;
149 if (max > size_)
150 max = size_;
151 if (min >= max || min >= size_) {
152 // Resulting slice is empty
153 data_.reset();
154 size_ = 0;
155 capacity_ = 0;
156 return;
157 }
158
159 int new_size = max - min;
160 int new_capacity = new_size;
161 std::unique_ptr<T[]> new_data = std::make_unique<T[]>(new_capacity);
162
163 // Move elements from the slice range to the new storage
164 for (int i = 0; i < new_size; ++i) {
165 new_data[i] = std::move(data_[min + i]);
166 }
167
168 // Update list state
169 data_ = std::move(new_data);
170 size_ = new_size;
171 capacity_ = new_capacity;
172}
